```javascript
function _ensureQnASheet(ss) {
  var sheet = ss.getSheetByName('QnA');
  if (!sheet) {
    sheet = ss.insertSheet('QnA');
    sheet.appendRow(['ID','Date','Name','Email','OrderCode','Category','Question','Answer','AnsweredAt','Status']);
```

```javascript
function getQnA(data) {
  var role = _getRole(data.password);
  if (role !== 'admin') return { error: '권한 없음' };
  var ss      = SpreadsheetApp.openById(SPREADSHEET_ID);
  var sheet   = _ensureQnASheet(ss);
  var rows    = sheet.getDataRange().getValues();
  var headers = rows[0];
  var threads = rows.slice(1).filter(function(r){ return r[0]; }).map(function(r){
    var o = {};
    headers.forEach(function(h, i){ o[h] = r[i]; });
    return o;
  });
  threads.sort(function(a, b){
    if (a.Status === 'PENDING' && b.Status !== 'PENDING') return -1;
    if (a.Status !== 'PENDING' && b.Status === 'PENDING') return 1;
    return b.Date > a.Date ? 1 : -1;
  });
  return { threads: threads };
}
```

```javascript
function getMyQnA(data) {
  if (!data.email) return { error: 'Email required' };
  var ss      = SpreadsheetApp.openById(SPREADSHEET_ID);
  var sheet   = _ensureQnASheet(ss);
  var rows    = sheet.getDataRange().getValues();
  var headers = rows[0];
  var threads = rows.slice(1)
    .filter(function(r){
      return r[0] && String(r[3]).toLowerCase() === data.email.toLowerCase();
    })
    .map(function(r){
      var o = {};
      headers.forEach(function(h, i){ o[h] = r[i]; });
      return o;
    });
  threads.sort(function(a, b){ return b.Date > a.Date ? 1 : -1; });
  return { threads: threads };
}
```

```javascript
function answerQnA(data) {
  var role = _getRole(data.password);
  if (role !== 'admin') return { error: '권한 없음' };
  if (!data.id || !data.answer) return { error: 'Missing id or answer' };

  var ss          = SpreadsheetApp.openById(SPREADSHEET_ID);
  var sheet       = _ensureQnASheet(ss);
  var rows        = sheet.getDataRange().getValues();
  var headers     = rows[0];
  var idIdx       = headers.indexOf('ID');
  var answerIdx   = headers.indexOf('Answer');
  var answeredIdx = headers.indexOf('AnsweredAt');
  var statusIdx   = headers.indexOf('Status');
  var answeredAt  = Utilities.formatDate(new Date(), 'Asia/Kolkata', 'yyyy-MM-dd HH:mm');

  for (var i = 1; i < rows.length; i++) {
    if (rows[i][idIdx] !== data.id) continue;
    sheet.getRange(i + 1, answerIdx   + 1).setValue(safe(data.answer));
    sheet.getRange(i + 1, answeredIdx + 1).setValue(answeredAt);
    sheet.getRange(i + 1, statusIdx   + 1).setValue('ANSWERED');
    var thread = {};
    headers.forEach(function(h, j){ thread[h] = rows[i][j]; });
    thread['Answer']     = data.answer;
    thread['AnsweredAt'] = answeredAt;
    sendQnAAnswerEmail(thread);
    return { success: true };
  }
  return { error: 'Thread not found' };
}
```

```javascript
// ── Q&A ──────────────────────────────────────────────────────
var qnaThreads = [];
var qnaFilter  = 'all';

function showQnA() {
  // Hide other sections — use the same pattern as existing showSection() calls
  document.querySelectorAll('[id$="Section"]').forEach(function(el){ el.style.display='none'; });
  document.getElementById('qnaSection').style.display = 'block';
  loadQnA();
}

async function loadQnA() {
  try {
    var res  = await post({ action:'getQnA', password: currentPassword });
    var data = await res.json();
    qnaThreads = data.threads || [];
    renderQnAList();
    var pending = qnaThreads.filter(function(t){ return t.Status==='PENDING'; }).length;
    var badge   = document.getElementById('qnaPendingCount');
    if (pending > 0) { badge.textContent = pending; badge.style.display='inline'; }
    else { badge.style.display='none'; }
  } catch(e) {
    document.getElementById('qnaList').innerHTML = '<div style="color:red;padding:12px">Failed to load Q&A</div>';
  }
}

function setQnAFilter(f, btn) {
  qnaFilter = f;
  document.querySelectorAll('.qna-filter-btn').forEach(function(b){ b.classList.remove('active'); });
  btn.classList.add('active');
  renderQnAList();
}

function renderQnAList() {
  var list     = document.getElementById('qnaList');
  var filtered = qnaFilter === 'all' ? qnaThreads : qnaThreads.filter(function(t){ return t.Status===qnaFilter; });
  if (!filtered.length) {
    list.innerHTML = '<div style="text-align:center;color:#AAA;padding:24px">No inquiries.</div>'; return;
  }
  list.innerHTML = filtered.map(function(t){
    return '<div class="qna-row" onclick="openQnAModal(\''+t.ID+'\')">'+
      '<div class="qna-row-meta">'+
        '<span class="qna-badge '+t.Status+'">'+(t.Status==='ANSWERED'?'Answered':'Pending')+'</span>'+
        '<span class="qna-cat">'+escA(t.Category)+'</span>'+
        '<span class="qna-date">'+escA(t.Date)+'</span>'+
      '</div>'+
      '<div class="qna-from">'+escA(t.Name)+' · '+escA(t.Email)+'</div>'+
      '<div class="qna-preview">'+escA(t.Question)+'</div>'+
    '</div>';
  }).join('');
}

function escA(s){ return String(s||'').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;'); }

function openQnAModal(id) {
  var t = qnaThreads.find(function(x){ return x.ID===id; });
  if (!t) return;
  document.getElementById('qnaModalContent').innerHTML =
    '<div style="margin-bottom:16px">'+
      '<div style="display:flex;align-items:center;gap:8px;margin-bottom:8px;flex-wrap:wrap">'+
        '<span class="qna-badge '+t.Status+'">'+(t.Status==='ANSWERED'?'Answered':'Pending')+'</span>'+
        '<span class="qna-cat">'+escA(t.Category)+'</span>'+
        '<span class="qna-date">'+escA(t.Date)+'</span>'+
      '</div>'+
      '<div style="font-size:13px;color:#644678;font-weight:600;margin-bottom:4px">'+escA(t.Name)+' · '+escA(t.Email)+(t.OrderCode?' · '+escA(t.OrderCode):'')+'</div>'+
      '<div style="font-size:14px;color:#1F1428;line-height:1.6;background:#F5EFE0;border-radius:8px;padding:12px">'+escA(t.Question).replace(/\n/g,'<br>')+'</div>'+
    '</div>'+
    (t.Answer
      ? '<div class="qna-existing-answer"><div style="font-size:10px;font-weight:700;color:#2E7D32;text-transform:uppercase;letter-spacing:.1em;margin-bottom:6px">Answered '+escA(t.AnsweredAt)+'</div>'+escA(t.Answer).replace(/\n/g,'<br>')+'</div>'
      : '<div class="qna-answer-wrap">'+
          '<div style="font-size:12px;font-weight:700;color:#444;margin-bottom:6px">Your Answer</div>'+
          '<textarea id="qnaAnswerText" placeholder="Type your reply…"></textarea>'+
          '<button class="qna-send-btn" onclick="sendQnAAnswer(\''+t.ID+'\')">Send Answer →</button>'+
        '</div>');
  document.getElementById('qnaModal').style.display = 'flex';
}

function closeQnAModal() { document.getElementById('qnaModal').style.display='none'; }

async function sendQnAAnswer(id) {
  var answer = document.getElementById('qnaAnswerText').value.trim();
  if (!answer) { alert('Please type an answer.'); return; }
  var btn = document.querySelector('.qna-send-btn');
  btn.textContent='Sending…'; btn.disabled=true;
  try {
    var res  = await post({ action:'answerQnA', password:currentPassword, id:id, answer:answer });
    var data = await res.json();
    if (data.success) { closeQnAModal(); await loadQnA(); }
    else { alert('Error: '+(data.error||'Unknown')); }
  } catch(e) { alert('Connection error.'); }
  btn.textContent='Send Answer →'; btn.disabled=false;
}
```
